  val h = list_head (x)
in
  list_cons (h, y)
end

val x1 = list_cons (5, list_nil)
val x2 = list_cons (6, list_nil)
val r = fn (x1, x2)

val x1 = list_cons ("a", list_nil)
val x2 = list_cons ("b", list_nil)

======================================
*)
